# 将环境变量替换配置文件

说明：经常会用到将环境变量中的值，替换项目的配置文件中的标记，利用简单的shell可以做到

PS：这里没有使用`envsubst`

## 配置文件格式

格式：{% raw %} `{{#变量名}}`、`{{#变量名-}}`、`{{#变量名-默认值}}` {% endraw %}

## shell
```bash
#!/bin/bash

# 初始化参数
input_file="config.json.env"
output_file="config.json"

# 解析命令行参数
while getopts "c:o:" opt; do
    case $opt in
        c)
            input_file="$OPTARG"
            ;;
        o)
            output_file="$OPTARG"
            ;;
        \?)
            echo "无效的选项: -$OPTARG" >&2
            exit 1
            ;;
    esac
done

# 检查输入文件是否存在
if [[ ! -f "$input_file" ]]; then
    echo "输入文件 '$input_file' 不存在。"
    exit 1
fi

# 读取 config.json.env 文件
while IFS= read -r line; do
    # 提取 ENV_NAME 和 DEFAULT_VALUE
    if [[ $line =~ (\{\{#([^}]+)\}\}) ]]; then
        all="${BASH_REMATCH[1]}"
        inner="${BASH_REMATCH[2]}"
        if [[ $inner =~ "-" ]]; then
            env_name="${inner%%-*}"
            default_value="${inner#*-}"
            if [[ $default_value = "" ]]; then
                value="${!env_name}"
            else
              value="${!env_name:-$default_value}"
            fi
        else
            value=${!inner}
        fi
        echo "${line/$all/$value}"
    else
        echo "$line"
    fi
done < "$input_file" > "$output_file"

echo "已生成 $output_file 文件。"
```
